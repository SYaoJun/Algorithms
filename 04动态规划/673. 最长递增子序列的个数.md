## 题意

-   考查知识点：dp

## 思路

-   `f[i]`表示以`a[i]`结尾的最大长度
-    `c[i]`表示以`a[i]`结尾的最大数量

## 代码

```cpp
class Solution {
public:
    int findNumberOfLIS(vector<int>& nums) {
        int n = nums.size();
        if(n <= 1) return n;
        vector<int> f(n,1), c(n,1);
        int len = 0;
        for(int i = 1; i < n; i++){
            for(int j = 0; j < i; j++){
                if(nums[j] < nums[i]) {
                    if(f[j]+1 > f[i]){
                        f[i] = f[j] + 1;
                        c[i] = c[j];
                    }else if(f[j]+1 == f[i]){
                        c[i] += c[j];
                    }
                }
            }
            len = max(len, f[i]);
        }
        int res = 0;
        for(int i = 0; i < n; i++){
            if(f[i] == len) res+= c[i];
        }
        return res;
    }
};
```
